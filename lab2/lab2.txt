bool LinkedList::del(char ch)
{
	Node * curr = head;
	Node * temp = curr;
	if(head->data == ch)
	{
		head = temp->next;
	}
	while(temp != nullptr && temp->data != ch)
	{
		temp = temp->next;
	}
	curr->next = temp->next;
	curr = curr->next;
	delete temp;
	return true;
}
